
4.-DISEÑO Y DESARROLLO DE ALGORITMOS Y ESQUEMAS DE BASE DE DATOS.

PROPÓSITO: SISTEMA DE GESTIÓN DE CLIENTES Y PRODUCTOS.

TABLAS:
CREATE TABLE CLIENTES(
ID_CLIENTE INTEGER(8) NOT NULL PRIMARY KEY,
NOMBRE VARCHAR(80),
DIRECCIÓN VARCHAR(170),
TELEFÓNO VARCHAR(10));

  CREATE TABLE PRODUCTOS(
  ID_PRODUCTO INTEGER(8) NOT NULL PRIMARY KEY,
  NOMBRE VARCHAR(90),
  PRECIO VARCHAR(10),
  STOCK CHAR(10));

VISTA:
CREATE VIEW ClientesActivos AS
SELECT * FROM Clientes WHERE id_cliente IN (
  SELECT DISTINCT id_cliente FROM Ventas WHERE fecha >= DATE_SUB(NOW(), INTERVAL 6 MONTH)
);

PROCESO:
CREATE PROCEDURE InsertarCliente(
    IN p_nombre VARCHAR(100),
    IN p_direccion VARCHAR(255),
    IN p_telefono VARCHAR(20)
)
BEGIN
    INSERT INTO Clientes (nombre, direccion, telefono) VALUES (p_nombre, p_direccion, p_telefono);
END;
